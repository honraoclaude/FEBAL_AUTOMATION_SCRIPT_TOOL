import hashlib, json
def cache_key(provider, model, messages, params) -> str:
    payload = json.dumps({"provider":provider,"model":model,"messages":_normalize(messages),
        "params":{"temperature":params.temperature,"max_tokens":params.max_tokens,"tools":params.tools}},
        sort_keys=True, separators=(",",":"))
    return "llm:cache:" + hashlib.sha256(payload.encode()).hexdigest()
